      campaign: 'launch'
    })
  },
  
  purchase_from_launch: (tier: string, amount: number) => {
    gtag('event', 'purchase', {
      transaction_id: Date.now().toString(),
      value: amount,
      currency: 'USD',
      campaign: 'launch',
      items: [{
        item_id: tier,
        item_name: tier,
        price: amount
      }]
    })
  },
  
  social_share: (platform: string) => {
    gtag('event', 'share', {
      method: platform,
      campaign: 'launch'
    })
  }
